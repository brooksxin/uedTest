dialog控件

普通对话框

打开对话框 关闭对话框

代码

			<a href="javascript:;" id="J_dialog_1">打开对话框</a>
			<a href="javascript:;" id="J_dialog_1_close">关闭对话框</a>

			$('#J_dialog_1').click(function(){
				var params = {
					id : 'd1',		//弹出对话框的id
					title : '提示信息',	//左上角提示标题
					content : 'hello world',	//具体提示内容
					width : '200',	//对话框宽度
					height : '50'	//对话框高度
				}
				Util.dialog.openDiv(params);
			});
			$('#J_dialog_1_close').click(function(){
				Util.dialog.close('d1');	//关闭对话框
			});

确认弹出框

打开对话框

代码


		<a href="javascript:;" id="J_dialog_3">打开对话框</a>

		$('#J_dialog_3').click(function(){
			var params = {
				id : 'd3',	//弹出对话框的id
				title : '提示信息',	//左上角提示标题
				content : '您确认要删除这条记录？',	//具体提示内容
				okVal: '确定',		//确认按钮文本
				okCallback: function () {	//确认按钮点击过之后执行的函数
			        this.title('提交中…');	//设置左上角标题
			        return false;
			    },
			    cancelVal: '取消',	//取消按钮文本
			    cancelCallback: function () {}	//取消按钮点击后执行的函数
			}
			Util.dialog.confirm(params);	//要调用确认对话框，记住confirm
		});

2秒后自动关闭弹出框

打开对话框

代码


		<a href="javascript:;" id="J_dialog_4">打开对话框</a>

		$('#J_dialog_4').click(function(){
			/*
			参数
				第一个 要显示的文本内容
				第二个 top对象，在iframe内层要弹窗到顶层时用到
				第三个 延迟关闭时间，单位（毫秒）
			*/
			Util.dialog.tips('hello world', null, 2000);	
			//也可简写如下，默认2秒后关闭
			Util.dialog.tips('hello world');
		})

气泡弹出框

打开对话框
目标

代码


		<a href="javascript:;" id="J_dialog_5">打开对话框</a>
		<div id="J_dialog_5_dest">目标</div>

		$('#J_dialog_5').click(function() {
			Util.dialog.bubble({ 
				element:$('#J_dialog_5_dest'), 	//要将气泡弹出在哪个元素上
				content:'欢迎使用气泡对话框', 	//气泡内容
				quickClose:true, 	//点击页面上其它位置是否关闭气泡 默认为true
				align: 'bottom right',	//气泡弹出方向，可以是 top、right、bottom、left、或者它们的组合，如top right
			});
		});
